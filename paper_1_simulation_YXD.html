def exp_prior(v0):
    return np.exp(-v0)

def normal_prior(v0, calibrate_y_series):
    log_price = calibrate_y_series
    if not isinstance(log_price, pd.Series):
        log_price = pd.Series(log_price)
    log_vol_series = np.log(log_price.rolling(20).var() * 250)
    avg = log_vol_series.mean()
    std = log_vol_series.std()
    return np.exp(np.exp(-(np.log(v0) - avg)**2 / (2 * std**2)) / (np.sqrt(2 * np.pi) * std))

def compute_vol(v0, y_series, theta, prior_func=exp_prior, T=None, delta=1/252, adj_overflow=True):
    """
    Compute the volatility and likelihood series using iterations.
    :param v0: array, initial volatility grid
    :param y_series: pd.Series/array-like, the return series, will be converted to pd.Series if not
    :param theta: array-like, list of parameters to be estimated
    :parma prior_func: callable, prior density function of L0
    :param T: int/None, default None, number of total iterations; if is None, refer to the length of y_series
    :param delta: float, default 1/252, length of time step
    :param adj_overflow: bool default True, if True, meaning Lt should be adjusted to avoid overflow,
                            divide by 1e30 every 50 iterations
    
    :return vol_res: np.array, volatility
    :return L_res: np.array, likelihood
    """
    if T is None:
        T = len(y_series)
    dim = v0.shape[0]
    mu, alpha, beta, sigma, rho = theta
    beta_prime = 1 - beta * delta
    D = 2 * np.pi * sigma * np.sqrt(1 - rho**2) * delta
    a = (beta_prime**2 + rho * sigma * beta_prime * delta + 0.25 * sigma**2 * delta**2) / (2 * sigma**2 * (1 - rho**2) * delta)
    if not isinstance(y_series, pd.Series):
        y_series = pd.Series(y_series)
    delta_y = y_series.diff().values
    vol_res, L_res = np.zeros((T, dim)), np.zeros((T, dim))
    vol_res[0] = v_pre = v0
    L_res[0] = L_pre = prior_func(v0)
    for t in range(1, T):
        delta_y_tmp = delta_y[t]
        N = delta_y_tmp - mu * delta  # this expression has been used repeatedly
        B = - alpha * delta - rho * sigma * N
        C = (alpha * delta)**2 + 2 * rho * sigma * alpha * delta * N + (sigma * N)**2 - 2 * v_pre**2 * a * sigma**2 * (1 - rho**2) * delta
        v_next = np.where(B**2-C>0, np.sqrt(B**2-C)-B, v_pre)
        M = v_next - alpha * delta  # this is also used repeatedly
        bt = (M**2 - 2 * rho * sigma * M * N + (sigma * N)**2) / (2 * sigma**2 * (1 - rho**2) * delta)
        dt_numerator = ((2 * beta_prime + rho * sigma * delta) * M - (2 * rho * sigma * beta_prime + sigma**2 * delta) * N) / (2 * sigma**2 * (1 - rho**2) * delta)
        if np.any(bt) < 0:
            print('bt<0 in iter: '+str(t), bt)
        L_next = np.exp(dt_numerator - 2 * np.sqrt(a*bt)) * np.power(a*bt, -0.25) * L_pre / D
        if adj_overflow and t % 50 == 0:
            L_next /= 1e30
        vol_res[t] = v_pre = v_next
        L_res[t] = L_pre = L_next
    return vol_res, L_res

def intgrt_likelihood(theta, v0_grid, y_series, adj_overflow):
    prior_func = lambda x: normal_prior(x, y_series)
    algo_vt, algo_Lt = compute_vol(v0_grid, y_series, theta, prior_func, adj_overflow=adj_overflow)
    integrate_x = algo_vt[-1]
    integrate_y = algo_Lt[-1][1:]
    res = np.nansum(np.diff(algo_vt[-1]) * algo_Lt[-1][1:])
#     print(res)
    return np.log(res)

# iterative function for paper 1, theta = (mu, alpha, beta, sigma, rho)
def compute_next(y, v, theta, sim_delta):
    mu, alpha, beta, sigma, rho = theta
    normal_mean = np.array([y + (mu - v / 2) * sim_delta, v + (alpha - beta * v) * sim_delta])
    normal_cov = np.array([[1, rho * sigma], [rho * sigma, sigma**2]]) * v * sim_delta
    y_next, v_next = np.random.multivariate_normal(normal_mean, normal_cov)
    return y_next, v_next


def simulation(y0, v0, theta, delta=1/252, n=252, sub_interval=30):
    sim_delta = delta / sub_interval
    res = np.zeros(shape=(n+1, 2))
    res[0] = (y0, v0)
    y = y0
    v = v0
    i = 0
    while i < int(n*sub_interval):
        y_tmp, v_tmp = compute_next(y, v, theta=theta, sim_delta=sim_delta)
        if v_tmp > 0:  # skip those simulated data where v <= 0
            i += 1
            y, v = y_tmp, v_tmp
            if i % sub_interval == 0:
                res[int(i/sub_interval)] = y, v
    sim_yt = res[:, 0]
    sim_vt = res[:, 1]
    return sim_yt, sim_vt

def sim_cost(theta):
    res = intgrt_likelihood(theta, v0_grid, sim_y, adj_overflow=True)
    return -res

def compute_descpt_stats(theta, sim_res):
    summary_res = pd.DataFrame(columns = ["True Value", "Estimate Mean", "Mean_Bias", "Estimate Standard Deviation"])
    for i in range(len(theta)):
        temp = [theta[i]]
        name = sim_res.columns[i]
        temp.append(sim_res[name].mean())
        temp.append(-theta[i]+sim_res[name].mean())
        temp.append(sim_res[name].std())
        summary_res = summary_res.append(dict(zip(summary_res.columns,temp)), ignore_index = True)
    summary_res.index = sim_res.columns[:len(theta)]
    return summary_res

def plot_hist(theta, sim_res):
    fig, axs = plt.subplots(1, len(theta), sharey=True, tight_layout=True,figsize=(3.5*len(theta),5))
    for i, name in enumerate(sim_res.columns[:len(theta)]):
        # We can set the number of bins with the `bins` kwarg
        axs[i].hist(sim_res[name], bins=50, label='opt_est')
        axs[i].title.set_text(name)
        axs[i].vlines(x=theta[i], ymin=axs[i].get_ylim()[0], ymax=axs[i].get_ylim()[1], linestyle='--', color='r',
                      label='true')
        axs[i].legend()
