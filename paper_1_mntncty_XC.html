    return 1/v0*1/(sigma*np.sqrt(2*np.pi))*np.exp(-(np.log(v0)-mu)**2/(2*sigma**2))+1

def normal_prior(v0, calibrate_y_series):
    log_price = calibrate_y_series
    if not isinstance(log_price, pd.Series):
        log_price = pd.Series(log_price)
    log_vol_series = np.log(log_price.rolling(20).var() * 250)
    avg = log_vol_series.mean()
    std = log_vol_series.std()
    return np.exp(np.exp(-(np.log(v0) - avg)**2 / (2 * std**2)) / (np.sqrt(2 * np.pi) * std))

def compute_vol(v0, y_series, theta, prior_func=compute_L0, T=250, delta=1/250):
    """
    Compute the volatility and likelihood series using iterations.
    :param v0: array, initial volatility grid
    :param y_series: pd.Series/array-like, the return series, will be converted to pd.Series if not
    :param theta: array-like, list of parameters to be estimated
    :parma prior_func: callable, prior density function of L0
    :param T: int, default 250, number of total iterations
    :param delta: float, default 1/250, length of time step, equal to 1/T
    
    :return vol_res: np.array, volatility
    :return L_res: np.array, likelihood
    """

    # delta = 1 / T
    dim = v0.shape[0]
    mu, alpha, beta, sigma, rho = theta
    beta_prime = 1 - beta * delta
    D = 2 * np.pi * sigma * np.sqrt(1 - rho**2) * delta
    a = (beta_prime**2 + rho * sigma * beta_prime * delta + 0.25 * sigma**2 * delta**2) / (2 * sigma**2 * (1 - rho**2) * delta)
    if not isinstance(y_series, pd.Series):
        y_series = pd.Series(y_series)
    delta_y = y_series.diff().values
    vol_res, L_res = np.zeros((T+1, dim)), np.zeros((T+1, dim))
    vol_res[0] = v_pre = v0
    L_res[0] = L_pre = prior_func(v0)
    
    for t in range(1, T+1):
        delta_y_tmp = delta_y[t]
        N = delta_y_tmp - mu * delta  # this expression has been used repeatedly
        B = - alpha * delta - rho * sigma * N
        C = (alpha * delta)**2 + 2 * rho * sigma * alpha * delta * N + (sigma * N)**2 - 2 * v_pre**2 * a * sigma**2 * (1 - rho**2) * delta
        
        v_next = np.sqrt(B**2 - C) - B
        
        M = v_next - alpha * delta  # this is also used repeatedly
        bt = (M**2 - 2 * rho * sigma * M * N + (sigma * N)**2) / (2 * sigma**2 * (1 - rho**2) * delta)
        dt_numerator = ((2 * beta_prime + rho * sigma * delta) * M - (2 * rho * sigma * beta_prime + sigma**2 * delta) * N) / (2 * sigma**2 * (1 - rho**2) * delta)

        L_next = np.exp(dt_numerator - 2 * np.sqrt(a*bt)) * np.power(a*bt, -0.25) * L_pre / D
        vol_res[t] = v_pre = v_next
        L_res[t] = L_pre = L_next
    return vol_res, L_res

Simulation of Heston Model¶

Related parameters and function in the simulation¶

  • Theta:
    $\theta = (\mu, \alpha, \beta, \sigma, \rho)^T = (0.025, 0.3, 3, 0.25, -0.8)^T$

  • Starting Points:
    $v_0 = \sqrt{0.001},y_0=7.5$

  • Generating Function: the transition probability density
    $p(y_{t+1},v_{t+1}|y_t,v_t) = N(\mu_{t+1},\Sigma_{t+1})$
    where
    $\mu_{t+1} = \left[ \begin{matrix} 
                 y_t+(\mu-\frac{v_t}{2}\Delta) \\
             v_t + (\alpha-\beta v_t)\Delta  
             \end{matrix} \right]$
    $\Sigma_{t+1} = v_t\Delta \left[ \begin{matrix} 
                                 1 & \rho\sigma\\
                             \rho\sigma & \sigma^2
                              \end{matrix} \right]$

def compute_next(y, v, theta, delta=1/250):
    mu, alpha, beta, sigma, rho = theta
    normal_mean = np.array([y + (mu - v / 2) * delta, v + (alpha - beta * v) * delta])
    normal_cov = np.array([[1, rho * sigma], [rho * sigma, sigma**2]]) * v * delta
    y_next, v_next = np.random.multivariate_normal(normal_mean, normal_cov)
    return y_next, v_next


def simulation(y0, v0, theta, delta=1/250):
    res = np.zeros(shape=(int(1/delta)+1, 2))
    res[0] = (y0, v0)
    y = y0
    v = v0
    i = 0
    while i < int(1/delta):
        y, v = compute_next(y, v, theta=theta, delta=delta)
        if v > 0:  # skip those simulated data where v <= 0
            res[i+1] = y, v
            i += 1
    yt_sim = res[:, 0]
    vt_sim = res[:, 1]
    
    return yt_sim, vt_sim

Apply the algorithm to the simulated path of v_t and y_t¶

Results

  • Picture1:
    We apply the algorithm to the simulated y_t and try to figure out the volatility path with the maximum likelihood. Compare the result with the simulated v_t, it is easy to see two paths are quite different, especially in the first half of the time.

  • Picture2: We generate the picture related to the relationship between starting point v0 and the likelihood at time n. With the passing of time, the peak is moving towards 0. The result does not change a lot when we try different prior function.

  • Picture2: Another small point is that due to the equation in calculating $v_{t+1}=\sqrt{B^2-C}-B$, $v_{t+1}$ could be NaN if $v_t$ is very small. So there is no result when v_t is very close to 0.

Possible Reasons about the problems in the algorithm

  • The simulated(real) v_t is not necessary the path with the maximum likelihood,
  • The time interval is not long enough for the calculation.
  • The result depends on the parameters we choose in the algorithm. For example, the starting point of v0, the length of time in total, the prior function of v0 and the theta we choose in the algorithm.

Parameter Optimization: Test of Monotonicity¶

Results: $\alpha,\sigma,\rho$ all have extreme points, however, not in the correct place! $\mu,\beta$ still show some form of monotonicity.¶

  • True Theta value for simulated data:
    $\mu = 0.025,\alpha=0.3,\beta=3,\sigma=0.25,\rho=-0.8$

  • Objective Functions:
    From the equation 35, $L_T = \int_{v_T}p(v_T,y_{1:T}|y_0,\theta)$
    Here we calculate the intergral using the formula: $\sum_{v_T}\Delta v_T * L_T(v_T)$

  • Observation: NaN values

def intgrt_likelihood(theta, v0, y_series):
    prior_func = lambda x: normal_prior(x, y_series)
    algo_vt, algo_Lt = compute_vol(v0, y_series, theta, prior_func)
    res = np.nansum(np.diff(algo_vt[-1]) * algo_Lt[-1][1:])
    return - np.log(res)
