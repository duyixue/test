def yahoo_parser(url):
    response = urlopen(url)
    data = json.loads(response.read().decode("utf-8"))['chart']['result'][0]
    t = pd.to_datetime(data['timestamp'], unit='s').date
    c = data['indicators']['quote'][0]['close']
    return pd.DataFrame({'close': c}, index=t)

def default_parser(url):
    return pd.read_csv(url, index_col=0, parse_dates=True)

def retrieve(source, code, label, column=None):
    query = {
        'yahoo': 'https://query1.finance.yahoo.com/v8/finance/chart/{}?period1=0&period2=5000000000&interval=1d',
    }
    parser = {
        'yahoo': yahoo_parser,
    }
    url = query.get(source).format(code)
    df = parser.get(source, default_parser)(url)
    df.index.name = None
    df.columns = [col.lower() for col in df.columns] if df.shape[1] > 1 else [label]
    if not column: column = df.columns
    all_dates = pd.date_range(df.index.min(), df.index.max(), freq='B')
    df = df.replace('.', None).reindex(all_dates).fillna(method='ffill')
    return df.astype('float')[column].squeeze()

def compute_likelihood(theta, y_series, vix_close, T=22/252, delta_t=1/252, vol_proxy='vix'):
    res = 0
    kai, gamma, sigma, rho = theta
    a, b = r-d,lambda1*(1-rho**2)+lambda2*rho-0.5
    kai_prime = kai+lambda2*sigma
    gamma_prime = gamma*kai/(kai+lambda2*sigma)
    a1, a2 = r - d, gamma_prime*kai_prime
    b1, b2 = rho*lambda2+(1-rho**2)*lambda1-0.5, lambda2*sigma-kai_prime
    a_prime = kai_prime * gamma_prime
    b_prime = -kai_prime
    
    x10 = y_series[0]
    if vol_proxy == 'vix':
        x20 = (vix_close[0]/100)**2
    elif vol_proxy == 'vix_exp':
        x20 = (b_prime*(vix_close[0]/100)**2+a_prime)*T/(np.exp(b_prime*(T))-1)-a_prime/b_prime
        if x20 <= 0: return -10000
    
    for t in range(1,len(vix_close)):
        x1, x10 = y_series[t],y_series[t-1]
        if vol_proxy == 'vix':
            x2 = (vix_close[t]/100)**2
            
        elif vol_proxy == 'vix_exp':
            x2 = (b_prime*(vix_close[t]/100)**2+a_prime)*T/(np.exp(b_prime*(T))-1)-a_prime/b_prime
            if x2 <= 0:
                return -10000
            
        Dv = 0.5*np.log((1-rho**2)*sigma*x2)
        
        ## Calculating the three order expansions
        c1 = -((x2-x20)**2-2*rho*sigma*(x2-x20)*(x1-x10)+sigma**2*(x1-x10)**2)/(2*(1-rho**2)*sigma**2*x20) + \
             (x2-x20)**3/(4*(1-rho**2)*sigma**2*x20**2) - (rho*(x2-x20)**2*(x1-x10))/(2*(1-rho**2)*sigma*x20**2) + \
             ((x2-x20)*(x1-x10)**2)/(4*(1-rho**2)*x20**2) + ((7*rho-8*rho**3)*(x2-x20)**3*(x1-x10))/(24*(1-rho**2)**2*sigma*x20**3) - \
             (7-10*rho**2)*(x2-x20)**2*(x1-x10)**2/(48*(1-rho**2)**2*x20**3) - \
             rho*sigma*(x2-x20)*(x1-x10)**3/(24*(1-rho**2)**2*x20**3) + sigma**2*(x1-x10)**4/(96*(1-rho**2)**2*x20**3) - \
             (15-16*rho**2)*(x2-x20)**4/(96*(1-rho**2)**2*sigma**2*x20**3)
        
        c2 = (x2-x20)*(x20*b2-rho*sigma*x20*b1-rho*sigma*a1+a2)/((1-rho**2)*sigma**2*x20) - \
             (x1-x10)*(rho*x20*b2-sigma*x20*b1-sigma*a1+rho*a2)/((1-rho**2)*sigma*x20) - sigma**2*(x1-x10)**2/(24*(1-rho**2)*x20**2) - \
             (x2-x20)**2*(sigma*(sigma-12*rho*a1)+12*a2)/(24*(1-rho**2)*sigma**2*x20**2) + (x2-x20)*(x1-x10)*(rho*sigma**2 - 6*sigma*a1+6*rho*a2)/(12*sigma*x20**2*(1-rho**2))
        
        c3 = (rho*sigma*a2*b1 + rho*sigma*a1*b2-sigma**2*a1*b1-a2*b2)/((1-rho**2)*sigma**2) + \
             (2*rho*sigma*b1*b2-sigma**2*b1**2-b2**2)*x20/(2*(1-rho**2)*sigma**2) - \
             (sigma**4-rho**2*sigma**4+6*sigma**2*a1**2 - 6*sigma**2*a2 + 6*rho**2*sigma**2*a2-12*rho*sigma*a1*a2+6*a2**2)/(12*(1-rho**2)*sigma**2*x20)
        temp = -0.5*m*np.log(2*np.pi*delta_t)-Dv+c1/delta_t + c2 + c3*delta_t
        
        x20 = x2
        x10 = x1
        
        res += temp
    return res

def compute_next(y, v, theta, sim_delta):
    kai, gamma, sigma, rho= theta
    a, b = r-d, lambda1*(1-rho**2)+lambda2*rho-0.5
    
    normal_mean = np.array([y + (a + b * v) * sim_delta, v + (kai * (gamma - v)) * sim_delta])
    normal_cov = np.array([[1, rho*sigma], [rho*sigma, sigma**2]]) * v * sim_delta
    y_next, v_next = np.random.multivariate_normal(normal_mean, normal_cov)
    return y_next, v_next


def simulation(y0, v0, theta, n = 252,delta=1/252, sub_interval=30):
    sim_delta = delta / sub_interval
    res = np.zeros(shape=(int(n+1), 2))
    res[0] = (y0, v0)
    y = y0
    v = v0
    i = 0
    while i < int(n*sub_interval):
        y_tmp, v_tmp = compute_next(y, v, theta=theta, sim_delta=sim_delta)
        if v_tmp > 0:  # skip those simulated data where v <= 0
            i += 1
            y, v = y_tmp, v_tmp
            if i % sub_interval == 0:
                res[int(i/sub_interval)] = y, v
    return res


def generate_sim_vix(theta, Yt, T = 30/365):
    kappa, gamma, sigma, rho = theta
    a_prime = kappa * gamma
    b_prime = -kappa
    sim_vix = (np.exp(b_prime*T)-1)*(Yt + a_prime/b_prime)/(b_prime*T)-a_prime/b_prime
    return sim_vix

def yahoo_parser(url):
    response = urlopen(url)
    data = json.loads(response.read().decode("utf-8"))['chart']['result'][0]
    t = pd.to_datetime(data['timestamp'], unit='s').date
    c = data['indicators']['quote'][0]['close']
    return pd.DataFrame({'close': c}, index=t)

def default_parser(url):
    return pd.read_csv(url, index_col=0, parse_dates=True)

def retrieve(source, code, label, column=None):
    query = {
        'yahoo': 'https://query1.finance.yahoo.com/v8/finance/chart/{}?period1=0&period2=5000000000&interval=1d',
    }
    parser = {
        'yahoo': yahoo_parser,
    }
    url = query.get(source).format(code)
    df = parser.get(source, default_parser)(url)
    df.index.name = None
    df.columns = [col.lower() for col in df.columns] if df.shape[1] > 1 else [label]
    if not column: column = df.columns
    all_dates = pd.date_range(df.index.min(), df.index.max(), freq='B')
    df = df.replace('.', None).reindex(all_dates).fillna(method='ffill')
    return df.astype('float')[column].squeeze()
